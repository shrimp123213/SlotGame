using System.Collections;

public interface ISkillUser
{
    bool CanMoveForward();
    IEnumerator MoveForward();

    IEnumerator PerformRangedAttack(TargetType targetType);
    IEnumerator IncreaseDefense(int value, TargetType targetType);

}
